115    public static final class Builder {
116
117        private final Options options = new Options();
118
119        private boolean validateTopicSyntax = true;
120
121        private final Set<GeneratorExtension> extensions = new HashSet<>();
122
123        private Builder() {
124        }
125
126        public void addExtension(final GeneratorExtension extension) {
127            Objects.requireNonNull(extension);
128
129            this.extensions.add(extension);
130        }
131
132        public Builder validateTopicSyntax(final boolean validateTopicSyntax) {
133            this.validateTopicSyntax = validateTopicSyntax;
134            return this;
135        }
136
137        public Builder targetPath(final Path targetPath) {
138            Objects.requireNonNull(targetPath);
139
140            this.options.targetPath = targetPath;
141            return this;
142        }
143
144        public Builder basePackage(final String basePackage) {
145            this.options.basePackage = basePackage;
146            return this;
147        }
148
149        public Builder characterSet(final Charset characterSet) {
150            this.options.characterSet = characterSet;
151            return this;
152        }
153
154        public Generator build(final AsyncApi api) {
155
156            final LinkedList<Exception> errors = new LinkedList<>();
157            this.options.validate(errors);
158
159            if (!errors.isEmpty()) {
160                final RuntimeException e = new RuntimeException("Invalid generator settings", errors.pollFirst());
161                errors.stream().forEach(e::addSuppressed);
162                throw e;
163            }
164
165            return new Generator(api, new Options(this.options), this.validateTopicSyntax, new ArrayList<>(this.extensions));
166        }
167    }

251    private void renderConnector(final ConnectorType connectorType) {
252
253        final TypeBuilder builder = createTypeBuilder();
254
255        final Consumer<TypeDeclaration> typeCustomizer = TypeBuilder.asInterface(true) //
256                .andThen(TypeBuilder.superInterfaces(Arrays.asList(TYPE_NAME_CONNECTOR)));
257
258        builder.createType(new TypeInformation(connectorType.getSimpleTypeName(), null, null), typeCustomizer, b -> {
259
260            renderDefaultConnectorBuilder(b, connectorType);
261
262            for (final Map.Entry<String, Map<String, List<Topic>>> versionEntry : this.serviceDefinitions.getVersions().entrySet()) {
263                final String version = makeVersion(versionEntry.getKey());
264
265                b.createType(new TypeInformation(version.toUpperCase(), null, null), true, false, vb -> {
266
267                    for (final Map.Entry<String, List<Topic>> serviceEntry : versionEntry.getValue().entrySet()) {
268
269                        final TypeInformation serviceType = createServiceTypeInformation(serviceEntry);
270
271                        vb.createMethod((ast, cu) -> {
272                            final MethodDeclaration md = ast.newMethodDeclaration();
273                            md.setName(ast.newSimpleName(PackageTypeBuilder.asPropertyName(serviceType.getName())));
274                            md.setReturnType2(ast.newSimpleType(ast.newName(packageName(connectorType.getPackageName(), version, serviceType.getName()))));
275                            return md;
276                        });
277
278                    }
279
280                });
281
282                // create version method - e.g. v1()
283
284                b.createMethod((ast, cu) -> {
285                    final MethodDeclaration md = ast.newMethodDeclaration();
286                    md.setName(ast.newSimpleName(version));
287                    md.setReturnType2(ast.newSimpleType(ast.newSimpleName(version.toUpperCase())));
288                    return md;
289                });
290
291            }
292
293            // create latest versions
294
295            for (final Map.Entry<String, VersionedService> latestEntry : this.serviceDefinitions.getLatest().entrySet()) {
296                b.createMethod((ast, cu) -> {
297                    return createReturnLatestVersionService(latestEntry, ast, connectorType);
298                });
299            }
300
301        });
302
303    }

306    private static ParameterizedType parametrizeSimple(final org.eclipse.jdt.core.dom.Type original, final String... parameters) {
307        final AST ast = original.getAST();
308
309        final ParameterizedType result = ast.newParameterizedType(original);
310
311        for (final String name : parameters) {
312            final SimpleType type = ast.newSimpleType(ast.newSimpleName(name));
313            result.typeArguments().add(type);
314        }
315
316        return result;
317    }

319    @SuppressWarnings("unchecked")
320    private void renderDefaultConnectorBuilder(final TypeBuilder builder, final ConnectorType connectorType) {
321
322        Consumer<TypeDeclaration> typeCustomizer = td -> {
323            final AST ast = td.getAST();
324
325            final SimpleType st = ast.newSimpleType(ast.newName(TYPE_NAME_ABSTRACT_CONNECTOR_BUILDER));
326
327            td.setSuperclassType(parametrizeSimple(st, "B", "C"));
328        };
329
330        typeCustomizer = typeCustomizer.andThen(td -> {
331            final AST ast = td.getAST();
332
333            final TypeParameter b = ast.newTypeParameter();
334            b.setName(ast.newSimpleName("B"));
335            b.typeBounds().add(parametrizeSimple(ast.newSimpleType(ast.newSimpleName("Builder")), "B", "C"));
336
337            final TypeParameter c = ast.newTypeParameter();
338            c.setName(ast.newSimpleName("C"));
339            c.typeBounds().add(ast.newSimpleType(ast.newSimpleName(connectorType.getSimpleTypeName())));
340
341            td.typeParameters().add(b);
342            td.typeParameters().add(c);
343        });
344
345        typeCustomizer = typeCustomizer.andThen(td -> JDTHelper.make(td, ModifierKeyword.STATIC_KEYWORD, ModifierKeyword.ABSTRACT_KEYWORD));
346
347        builder.createType(new TypeInformation("Builder", null, null), typeCustomizer, b -> {
348
349            b.createMethod((ast, cu) -> {
350                final MethodDeclaration md = ast.newMethodDeclaration();
351                md.setConstructor(true);
352                md.setName(ast.newSimpleName("Builder"));
353                makeProtected(md);
354
355                final Block body = ast.newBlock();
356                md.setBody(body);
357
358                if (this.api.getHost() != null && !this.api.getHost().isEmpty()) {
359                    final MethodInvocation mi = ast.newMethodInvocation();
360                    mi.setName(ast.newSimpleName("host"));
361                    mi.arguments().add(newStringLiteral(ast, this.api.getHost()));
362                    body.statements().add(ast.newExpressionStatement(mi));
363                }
364
365                if (this.api.getBaseTopic() != null && !this.api.getBaseTopic().isEmpty()) {
366                    final MethodInvocation mi = ast.newMethodInvocation();
367                    mi.setName(ast.newSimpleName("baseTopic"));
368                    mi.arguments().add(newStringLiteral(ast, this.api.getBaseTopic()));
369                    body.statements().add(ast.newExpressionStatement(mi));
370                }
371
372                return md;
373            });
374
375        });
376    }

378    @SuppressWarnings("unchecked")
379    private MethodDeclaration createReturnLatestVersionService(final Map.Entry<String, VersionedService> latestEntry, final AST ast, final ConnectorType connectorType) {
380        final String version = makeVersion(latestEntry.getValue().getVersion().toString());
381        final String serviceType = packageName(connectorType.getPackageName(), version, latestEntry.getValue().getType().getName());
382
383        final MethodDeclaration md = ast.newMethodDeclaration();
384
385        md.setName(ast.newSimpleName(asPropertyName(latestEntry.getKey())));
386        md.setReturnType2(ast.newSimpleType(ast.newName(serviceType)));
387
388        md.modifiers().add(ast.newModifier(ModifierKeyword.DEFAULT_KEYWORD));
389
390        final Block block = ast.newBlock();
391        md.setBody(block);
392
393        final ReturnStatement ret = ast.newReturnStatement();
394        block.statements().add(ret);
395
396        final MethodInvocation versionMethod = ast.newMethodInvocation();
397        versionMethod.setName(ast.newSimpleName(version.toLowerCase()));
398
399        final MethodInvocation serviceMethod = ast.newMethodInvocation();
400        serviceMethod.setName(ast.newSimpleName(asPropertyName(latestEntry.getKey())));
401        serviceMethod.setExpression(versionMethod);
402
403        ret.setExpression(serviceMethod);
404
405        return md;
406    }

408    @SuppressWarnings("unchecked")
409    private void renderServices(final ConnectorType connectorType) {
410        for (final Map.Entry<String, Map<String, List<Topic>>> versionEntry : this.serviceDefinitions.getVersions().entrySet()) {
411
412            final String packageName = connectorType.getPackageName();
413            final String version = makeVersion(versionEntry.getKey());
414            final TypeBuilder builder = new PackageTypeBuilder(this.options.getTargetPath(), packageName(packageName, version), this.options.getCharacterSet(), type -> null,
415                    this::lookupType);
416
417            for (final Map.Entry<String, List<Topic>> serviceEntry : versionEntry.getValue().entrySet()) {
418                builder.createType(createServiceTypeInformation(serviceEntry), true, false, b -> {
419
420                    for (final Topic topic : serviceEntry.getValue()) {
421                        b.createMethod((ast, cu) -> {
422
423                            final TopicInformation ti = this.serviceDefinitions.getTopics().get(topic);
424
425                            // new method
426
427                            final MethodDeclaration md = ast.newMethodDeclaration();
428                            md.setName(ast.newSimpleName(makeTopicMethodName(ti)));
429
430                            // return type
431
432                            // set return type
433
434                            md.setReturnType2(evalEventMethodType(ast, topic, this.context, connectorType));
435
436                            // assign annotation
437
438                            final NormalAnnotation an = ast.newNormalAnnotation();
439                            an.setTypeName(ast.newName(TYPE_NAME_TOPIC_ANN));
440                            an.values().add(newKeyValueString(ast, "name", topic.getName()));
441                            if (topic.getPublish() != null) {
442                                an.values().add(newKeyValueClass(ast, "publish", messageTypeName(topic.getPublish(), this.context)));
443                            }
444                            if (topic.getSubscribe() != null) {
445                                an.values().add(newKeyValueClass(ast, "subscribe", messageTypeName(topic.getSubscribe(), this.context)));
446                            }
447                            md.modifiers().add(an);
448
449                            // make public
450
451                            makePublic(md);
452
453                            // return
454
455                            return md;
456
457                        });
458
459                    }
460
461                });
462            }
463
464        }
465    }

471    @SuppressWarnings("unchecked")
472    public static ParameterizedType evalEventMethodType(final AST ast, final Topic topic, final Context context, final ConnectorType connectorType) {
473
474        final MessageReference pubMsg = connectorType.getPublish(topic);
475        final MessageReference subMsg = connectorType.getSubscribe(topic);
476
477        if (pubMsg == null && subMsg == null) {
478            return null;
479        }
480
481        final SimpleType eventType;
482
483        if (pubMsg != null && subMsg != null) {
484            eventType = ast.newSimpleType(ast.newName(TYPE_NAME_PUBSUB_CLASS));
485        } else if (pubMsg != null) {
486            eventType = ast.newSimpleType(ast.newName(TYPE_NAME_PUB_CLASS));
487        } else {
488            eventType = ast.newSimpleType(ast.newName(TYPE_NAME_SUB_CLASS));
489        }
490
491        final ParameterizedType type = ast.newParameterizedType(eventType);
492
493        if (pubMsg != null) {
494            type.typeArguments().add(ast.newSimpleType(ast.newName(messageTypeName(pubMsg, context))));
495        }
496        if (subMsg != null) {
497            type.typeArguments().add(ast.newSimpleType(ast.newName(messageTypeName(subMsg, context))));
498        }
499
500        return type;
501    }

535    public static String makeTopicMethodName(final TopicInformation ti) {
536
537        Stream<String> s = Stream.of(ti.getType());
538
539        s = Stream.concat(s, ti.getResources().stream());
540        s = Stream.concat(s, Stream.of(ti.getAction()));
541
542        if (ti.getStatus().isPresent()) {
543            s = Stream.concat(s, Stream.of(ti.getStatus().get().toString().toLowerCase()));
544        }
545
546        return joinLowerCamelCase(s);
547    }

549    private static String joinLowerCamelCase(final Stream<String> s) {
550
551        boolean first = true;
552
553        final StringBuilder sb = new StringBuilder();
554
555        for (final String tok : (Iterable<String>) s::iterator) {
556            final int len = tok.length();
557
558            if (len == 0) {
559                continue;
560            }
561
562            final Function<Character, Character> fn = first ? Character::toLowerCase : Character::toUpperCase;
563            first = false;
564
565            sb.append(fn.apply(tok.charAt(0)));
566            sb.append(tok.substring(1).toLowerCase());
567        }
568
569        return sb.toString();
570    }

586    private void generateMessage(final TypeBuilder builder, final Message message) {
587
588        final TypeInformation ti = new TypeInformation(asTypeName(message.getName()), message.getSummary(), message.getDescription());
589
590        //         final String payloadTypeName = PackageTypeBuilder.asTypeName(message.getPayload().getName());
591
592        @SuppressWarnings("unchecked")
593        final Consumer<TypeDeclaration> typeCustomizer = td -> {
594            final AST ast = td.getAST();
595
596            final ParameterizedType type = ast.newParameterizedType(ast.newSimpleType(ast.newName(TYPE_NAME_MESSAGE_INTERFACE)));
597            type.typeArguments().add(ast.newSimpleType(ast.newName(ti.getName() + ".Payload")));
598
599            td.superInterfaceTypes().add(type);
600        };
601
602        final TypeReference payloadType = message.getPayload();
603
604        builder.createType(ti, typeCustomizer, b -> {
605
606            if (payloadType instanceof ObjectType) {
607
608                generateType(b, (Type) payloadType);
609
610                b.createProperty(new PropertyInformation((Type) payloadType, "payload", "Message payload", null));
611
612            } else if (payloadType instanceof CoreType) {
613
614                b.createProperty(new PropertyInformation((CoreType) message.getPayload(), "payload", "Message payload", null));
615
616            } else if (payloadType.getClass().equals(TypeReference.class)) {
617                b.createProperty(new PropertyInformation(lookupType(payloadType), "payload", "Message payload", null));
618            } else {
619                throw new IllegalStateException("Unsupported payload type: " + message.getPayload().getClass().getName());
620            }
621
622        });
623    }

656    private static String toPrimitives(final CoreType type) {
657        final String typeName = type.getJavaType().getName();
658
659        switch (typeName) {
660        case "java.lang.Boolean":
661            return "boolean";
662        case "java.lang.Integer":
663            return "int";
664        case "java.lang.Long":
665            return "long";
666        case "java.lang.Float":
667            return "float";
668        case "java.lang.Short":
669            return "short";
670        case "java.lang.Byte":
671            return "byte";
672        case "java.lang.Character":
673            return "char";
674        case "java.lang.Double":
675            return "double";
676        default:
677            return typeName;
678        }
679
680    }

686    private String resolveTypeName(final TypeReference typeRef, final boolean allowPrimitives) {
687        if (typeRef instanceof ObjectType || typeRef instanceof EnumType) {
688
689            return resolveParentableTypeName(typeRef);
690
691        } else if (typeRef instanceof CoreType) {
692
693            if (allowPrimitives) {
694                return toPrimitives((CoreType) typeRef);
695            } else {
696                return ((CoreType) typeRef).getJavaType().getName();
697            }
698
699        } else if (typeRef instanceof ArrayType) {
700
701            return resolveTypeName(((ArrayType) typeRef).getItemType(), false);
702
703        } else {
704
705            return resolveTypeName(lookupType(typeRef.getName()), allowPrimitives);
706
707        }
708    }

710    private String resolveParentableTypeName(final TypeReference typeRef) {
711        final List<String> full = new LinkedList<>();
712
713        full.add(typeRef.getNamespace());
714
715        if (typeRef instanceof ParentableType) {
716            for (final String parent : ((ParentableType) typeRef).getParents()) {
717                full.add(asTypeName(parent));
718            }
719        }
720
721        full.add(asTypeName(typeRef.getName()));
722
723        return packageName(String.join(".", full));
724    }
725
726    private Type lookupType(final TypeReference typeRef) {
727        if (typeRef instanceof Type) {
728            return (Type) typeRef;
729        }
730
731        return lookupType(typeRef.getName());
732    }

753    private void generateEnum(final EnumType type, final TypeBuilder builder) {
754        builder.createEnum(new TypeInformation(asTypeName(type.getName()), type.getTitle(), type.getDescription()), type.getLiterals(),
755                (literal, decl) -> fireExtensions(extension -> extension.createdEnumLiteral(literal, decl)), true);
756    }

758    protected <T> void fireExtensions(final Consumer<GeneratorExtension> consumer) {
759        for (final GeneratorExtension extension : this.extensions) {
760            consumer.accept(extension);
761        }
762    }

764    protected <T> void fireExtensions(final T literal, final BiConsumer<GeneratorExtension, T> consumer) {
765        for (final GeneratorExtension extension : this.extensions) {
766            consumer.accept(extension, literal);
767        }
768    }

770    private String packageName(final String... local) {
771
772        Stream<String> full;
773
774        if (this.options.getBasePackage() != null && !this.options.getBasePackage().isEmpty()) {
775            full = Stream.of(this.options.getBasePackage());
776        } else if (this.api.getBaseTopic() != null && !this.api.getBaseTopic().isEmpty()) {
777            full = Stream.of(this.api.getBaseTopic());
778        } else {
779            full = Stream.empty();
780        }
781
782        if (local != null) {
783            full = Stream.concat(full, Arrays.stream(local));
784        }
785
786        return full.collect(Collectors.joining("."));
787    }

789    @SuppressWarnings("unchecked")
790    private void generateRoot() throws IOException {
791
792        PackageTypeBuilder.createCompilationUnit(this.options.getTargetPath(), packageName(), "package-info", this.options.getCharacterSet(), (ast, cu) -> {
793            final Information info = this.api.getInformation();
794
795            final Javadoc doc = ast.newJavadoc();
796
797            if (info.getTitle() != null) {
798                final TagElement tag = ast.newTagElement();
799                tag.fragments().add(JDTHelper.newText(ast, info.getTitle()));
800                doc.tags().add(tag);
801            }
802
803            if (info.getVersion() != null) {
804                final TagElement version = ast.newTagElement();
805                version.setTagName("@version");
806                version.fragments().add(JDTHelper.newText(ast, info.getVersion()));
807                doc.tags().add(version);
808            }
809
810            cu.getPackage().setJavadoc(doc);
811
812        });
813    }
