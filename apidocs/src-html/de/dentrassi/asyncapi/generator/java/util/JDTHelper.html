017package de.dentrassi.asyncapi.generator.java.util;
018
019import java.util.List;
020
021import org.eclipse.jdt.core.dom.AST;
022import org.eclipse.jdt.core.dom.Block;
023import org.eclipse.jdt.core.dom.BodyDeclaration;
024import org.eclipse.jdt.core.dom.CatchClause;
025import org.eclipse.jdt.core.dom.Expression;
026import org.eclipse.jdt.core.dom.FieldDeclaration;
027import org.eclipse.jdt.core.dom.MarkerAnnotation;
028import org.eclipse.jdt.core.dom.Modifier.ModifierKeyword;
029import org.eclipse.jdt.core.dom.SingleVariableDeclaration;
030import org.eclipse.jdt.core.dom.StringLiteral;
031import org.eclipse.jdt.core.dom.TextElement;
032import org.eclipse.jdt.core.dom.Type;
033import org.eclipse.jdt.core.dom.VariableDeclarationFragment;
034
035@SuppressWarnings("unchecked")
036public final class JDTHelper {
037    private JDTHelper() {
038    }
039
040    public static TextElement newText(final AST ast, final String text) {
041        final TextElement result = ast.newTextElement();
042        result.setText(text);
043        return result;
044    }
045
046    public static StringLiteral newStringLiteral(final AST ast, final String value) {
047        final StringLiteral result = ast.newStringLiteral();
048        result.setLiteralValue(value);
049        return result;
050    }
051
052    public static void make(final AST ast, final List<? super Object> decl, final ModifierKeyword... keywords) {
053        for (final ModifierKeyword keyword : keywords) {
054            decl.add(ast.newModifier(keyword));
055        }
056    }
057
058    public static void make(final BodyDeclaration decl, final ModifierKeyword... keywords) {
059        make(decl.getAST(), decl.modifiers(), keywords);
060    }
061
062    public static void make(final SingleVariableDeclaration decl, final ModifierKeyword... keywords) {
063        make(decl.getAST(), decl.modifiers(), keywords);
064    }
065
066    public static void makeStatic(final BodyDeclaration decl) {
067        make(decl, ModifierKeyword.STATIC_KEYWORD);
068    }
069
070    public static void makeAbstract(final BodyDeclaration decl) {
071        make(decl, ModifierKeyword.ABSTRACT_KEYWORD);
072    }
073
074    public static void makePublic(final BodyDeclaration decl) {
075        make(decl, ModifierKeyword.PUBLIC_KEYWORD);
076    }
077
078    public static void makeProtected(final BodyDeclaration decl) {
079        make(decl, ModifierKeyword.PROTECTED_KEYWORD);
080    }
081
082    public static void makePrivate(final BodyDeclaration decl) {
083        make(decl, ModifierKeyword.PRIVATE_KEYWORD);
084    }
085
086    public static SingleVariableDeclaration createParameter(final AST ast, final String typeName, final String name, final ModifierKeyword... keywords) {
087        final SingleVariableDeclaration arg = ast.newSingleVariableDeclaration();
088        arg.setName(ast.newSimpleName(name));
089        arg.setType(ast.newSimpleType(ast.newName(typeName)));
090        make(arg, keywords);
091        return arg;
092    }
093
094    public static FieldDeclaration createField(final AST ast, final String typeName, final String name, final ModifierKeyword... keywords) {
095        return createField(ast, ast.newSimpleType(ast.newName(typeName)), name, null, keywords);
096    }
097
098    public static FieldDeclaration createField(final AST ast, final Type type, final String name, final Expression initializer, final ModifierKeyword... keywords) {
099        final VariableDeclarationFragment vdf = ast.newVariableDeclarationFragment();
100        vdf.setName(ast.newSimpleName(name));
101        if (initializer != null) {
102            vdf.setInitializer(initializer);
103        }
104        final FieldDeclaration fd = ast.newFieldDeclaration(vdf);
105        make(fd, keywords);
106        fd.setType(type);
107        return fd;
108    }
109
110    public static void addSimpleAnnotation(final BodyDeclaration decl, final String name) {
111        final AST ast = decl.getAST();
112        final MarkerAnnotation ann = ast.newMarkerAnnotation();
113        ann.setTypeName(ast.newName(name));
114        decl.modifiers().add(ann);
115    }
116
117    public static CatchClause createCatchBlock(final AST ast, final String typeName, final Block catchBlock) {
118        final CatchClause cc = ast.newCatchClause();
119        cc.setException(JDTHelper.createParameter(ast, typeName, "e", ModifierKeyword.FINAL_KEYWORD));
120        cc.setBody(catchBlock);
121        return cc;
122    }
123
124}
