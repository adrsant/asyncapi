017package de.dentrassi.asyncapi.generator.java.util;
018
019import java.util.Collections;
020import java.util.List;
021import java.util.function.Function;
022import java.util.stream.Stream;
023
024import org.eclipse.jdt.core.dom.AST;
025import org.eclipse.jdt.core.dom.ASTNode;
026import org.eclipse.jdt.core.dom.ASTParser;
027import org.eclipse.jdt.core.dom.TypeDeclaration;
028
029public final class Java {
030    private Java() {
031    }
032
033    /**
034     * Stream a list of {@link ASTNode}s
035     *
036     * @param nodes
037     *            the nodes to stream
038     * @return the stream
039     */
040    @SuppressWarnings("unchecked")
041    public static Stream<ASTNode> stream(final List<?> nodes) {
042        return ((List<ASTNode>) nodes).stream();
043    }
044
045    /**
046     * Stream a list of {@link ASTNode}s from an {@link ASTNode} property
047     * <p>
048     * This can be used e.g. like the following:
049     * </p>
050     * 
051     * <pre>
052     * stream(node, TypeDeclaration.class, TypeDeclaration::bodyDeclaration)
053     *         .findFirst()
054     *         .get();
055     * </pre>
056     *
057     * @param node
058     *            the source node
059     * @param clazz
060     *            the required type of the source node
061     * @param supplier
062     *            the supplier of nodes
063     * @param <T>
064     *            the type of the {@link ASTNode}
065     * @return a stream of nodes
066     */
067    public static <T extends ASTNode> Stream<ASTNode> stream(final ASTNode node, final Class<T> clazz, final Function<T, List<?>> supplier) {
068        return stream(supplier.apply(clazz.cast(node)));
069    }
070
071    /**
072     * Get the first node of a {@link TypeDeclaration#bodyDeclarations()}
073     * <p>
074     * <b>Note: </b> Fails of there is no first node.
075     * </p>
076     *
077     * @param node
078     *            the node to work on
079     * @return the first node from the body declaration
080     */
081    public static ASTNode firstBodyDeclaration(final ASTNode node) {
082        return stream(node, TypeDeclaration.class, TypeDeclaration::bodyDeclarations)
083                .findFirst()
084                .get();
085    }
086
087    @SuppressWarnings("unchecked")
088    public static List<ASTNode> parse(final AST ast, final int kind, final String code, final Function<ASTNode, List<ASTNode>> extractor) {
089        final List<ASTNode> result = parseInternal(ast, kind, code, node -> {
090            if (extractor != null) {
091                return extractor.apply(node);
092            } else {
093                return Collections.singletonList(node);
094            }
095        });
096
097        return ASTNode.copySubtrees(ast, result);
098    }
099
100    public static ASTNode parseSingle(final AST ast, final int kind, final String code, final Function<ASTNode, ASTNode> extractor) {
101        final ASTNode result = parseInternal(ast, kind, code, node -> {
102            if (extractor != null) {
103                return extractor.apply(node);
104            } else {
105                return node;
106            }
107        });
108
109        return ASTNode.copySubtree(ast, result);
110    }
111
112    public static List<ASTNode> parseSingleList(final AST ast, final int kind, final String code, final Function<ASTNode, ASTNode> extractor) {
113        final ASTNode result = parseSingle(ast, kind, code, extractor);
114
115        if (result == null) {
116            throw new IllegalStateException("No element parsed");
117        }
118
119        return Collections.singletonList(result);
120    }
121
122    public static <T> T parseInternal(final AST ast, final int kind, final String code, final Function<ASTNode, T> extractor) {
123        final ASTParser parser = ASTParser.newParser(AST.JLS8);
124
125        parser.setKind(kind);
126        parser.setSource(code.toCharArray());
127
128        final ASTNode node = parser.createAST(null);
129
130        return extractor.apply(node);
131    }
132
133}
