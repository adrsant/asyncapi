017package de.dentrassi.asyncapi.jms;
018
019import java.io.Serializable;
020import java.util.LinkedList;
021import java.util.Objects;
022import java.util.concurrent.CompletableFuture;
023import java.util.concurrent.Executor;
024import java.util.function.Consumer;
025
026import javax.jms.Connection;
027import javax.jms.Destination;
028import javax.jms.JMSException;
029import javax.jms.Message;
030import javax.jms.MessageConsumer;
031import javax.jms.Session;
032
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035
036import de.dentrassi.asyncapi.ListenerHandle;
037import de.dentrassi.asyncapi.Subscribe;
038
039public class JmsSubscriber<T extends de.dentrassi.asyncapi.Message<P>, P extends Serializable> implements Subscribe<T> {
040
041    private static final Logger logger = LoggerFactory.getLogger(JmsSubscriber.class);
042
043    private class HandleImpl extends CompletableFuture<Void> implements ListenerHandle {
044
045        private Session session;
046        private MessageConsumer consumer;
047        private final Consumer<T> handler;
048
049        public HandleImpl(final Consumer<T> consumer) {
050            this.handler = consumer;
051        }
052
053        @Override
054        public void close() throws Exception {
055
056            try {
057                get();
058            } catch (final Exception e) {
059                // ignore
060            }
061
062            LinkedList<Exception> errors = null;
063
064            if (this.consumer != null) {
065                try {
066                    this.consumer.close();
067                } catch (final Exception e) {
068                    if (errors == null) {
069                        errors = new LinkedList<>();
070                    }
071                    errors.add(e);
072                }
073            }
074            if (this.session != null) {
075                try {
076                    this.session.close();
077                } catch (final Exception e) {
078                    if (errors == null) {
079                        errors = new LinkedList<>();
080                    }
081                    errors.add(e);
082                }
083            }
084
085            if (errors != null && !errors.isEmpty()) {
086                final Exception e = errors.pollFirst();
087                errors.stream().forEach(e::addSuppressed);
088                throw e;
089            }
090        }
091
092        public void subscribe() {
093            try {
094                this.session = JmsSubscriber.this.connection.createSession(Session.CLIENT_ACKNOWLEDGE);
095                final Destination destination = this.session.createTopic(JmsSubscriber.this.topic);
096                this.consumer = this.session.createConsumer(destination);
097                this.consumer.setMessageListener(this::processMessage);
098
099                complete(null);
100            } catch (final Exception e) {
101                completeExceptionally(e);
102            }
103        }
104
105        protected void processMessage(final Message message) {
106            logger.debug("Received message: {}", message);
107
108            try {
109                final T m = JmsSubscriber.this.payloadFormat.decode(JmsSubscriber.this.clazz, JmsSubscriber.this.payloadClazz, message);
110                if (m != null) {
111                    this.handler.accept(m);
112                }
113                message.acknowledge();
114            } catch (final Exception e) {
115                logger.debug("Failed to handle message", e);
116                try {
117                    this.session.recover();
118                } catch (final JMSException e1) {
119                    // FIXME: we need to handle this somehow
120                    // possible solution, re-create session
121                }
122            }
123        }
124
125    };
126
127    private final Class<T> clazz;
128    private final Class<P> payloadClazz;
129    private final JmsPayloadFormat payloadFormat;
130    private final String topic;
131    private final Connection connection;
132    private final Executor executor;
133
134    public JmsSubscriber(final Class<T> clazz, final Class<P> payloadClazz, final JmsPayloadFormat payloadFormat, final String topic, final Connection connection,
135            final Executor executor) {
136        this.clazz = clazz;
137        this.payloadClazz = payloadClazz;
138        this.payloadFormat = payloadFormat;
139        this.topic = topic;
140        this.connection = connection;
141        this.executor = executor;
142    }
143
144    @Override
145    public ListenerHandle subscribe(final Consumer<T> consumer) {
146        Objects.requireNonNull(consumer);
147
148        final HandleImpl handle = new HandleImpl(consumer);
149
150        this.executor.execute(() -> {
151            handle.subscribe();
152        });
153
154        return handle;
155    }
156
157}
